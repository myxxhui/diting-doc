| 数据类 | 主存 | 索引 | 归档 |

| 模板 | PostgreSQL | — | OSS（按版本） |

## 二、表结构（PostgreSQL）

### 2.1 templates

```sql
CREATE TABLE templates (
  template_id TEXT NOT NULL,
  template_version TEXT NOT NULL,         -- SemVer
  schema_version TEXT NOT NULL,
  description TEXT,
  nodes JSONB NOT NULL,
  probes JSONB NOT NULL,
  transitions JSONB NOT NULL,
  guards JSONB,
  advisories JSONB,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  PRIMARY KEY (template_id, template_version)
);

CREATE INDEX idx_templates_id ON templates(template_id);
```

### 2.2 instances

```sql
CREATE TABLE instances (
  instance_id UUID PRIMARY KEY,
  template_id TEXT NOT NULL,
  template_version TEXT NOT NULL,
  schema_version TEXT NOT NULL,
  subject_id TEXT NOT NULL,
  subject_type TEXT NOT NULL,
  current_state TEXT NOT NULL,
  state_entered_at TIMESTAMPTZ NOT NULL,
  context JSONB,
  owner_user_id TEXT NOT NULL,
  group_id TEXT,
  status TEXT NOT NULL DEFAULT 'ACTIVE',  -- ACTIVE / SUSPENDED / ARCHIVED / CORRUPTED
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  FOREIGN KEY (template_id, template_version) REFERENCES templates(template_id, template_version)
);

CREATE INDEX idx_instances_subject_status ON instances(subject_id, status);
CREATE INDEX idx_instances_owner_status ON instances(owner_user_id, status);
CREATE INDEX idx_instances_group ON instances(group_id);
CREATE INDEX idx_instances_template ON instances(template_id, template_version);
```

### 2.3 instance_history

```sql
CREATE TABLE instance_history (
  id UUID PRIMARY KEY,
  instance_id UUID NOT NULL REFERENCES instances(instance_id),
  from_state TEXT NOT NULL,
  to_state TEXT NOT NULL,
  transition_event_id UUID NOT NULL,
  at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_history_instance_at ON instance_history(instance_id, at DESC);
```

### 2.4 probe_results（TimescaleDB hypertable 推荐）

```sql
CREATE TABLE probe_results (
  id UUID NOT NULL,
  probe_id TEXT NOT NULL,
  instance_id UUID NOT NULL,
  subject_id TEXT NOT NULL,
  payload JSONB NOT NULL,
  source_ref TEXT,
  degraded BOOLEAN NOT NULL DEFAULT FALSE,
  captured_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  PRIMARY KEY (id, captured_at)
);

-- TimescaleDB
SELECT create_hypertable('probe_results', 'captured_at', chunk_time_interval => INTERVAL '1 day');

CREATE INDEX idx_probe_instance_captured ON probe_results(instance_id, captured_at DESC);
CREATE INDEX idx_probe_probe_captured ON probe_results(probe_id, captured_at DESC);
```

### 2.5 transition_events

```sql
CREATE TABLE transition_events (
  id UUID PRIMARY KEY,
  instance_id UUID NOT NULL REFERENCES instances(instance_id),
  template_id TEXT NOT NULL,
  template_version TEXT NOT NULL,
  from_state TEXT NOT NULL,
  to_state TEXT NOT NULL,
  trigger_probe_id TEXT NOT NULL,
  trigger_probe_result_id UUID NOT NULL,
  evaluated_conditions JSONB NOT NULL,
  transitioned_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
) PARTITION BY RANGE (transitioned_at);

CREATE INDEX idx_transitions_instance_at ON transition_events(instance_id, transitioned_at DESC);
CREATE INDEX idx_transitions_template_at ON transition_events(template_id, transitioned_at DESC);
```

### 2.6 boundary_breaches

```sql
CREATE TABLE boundary_breaches (
  id UUID PRIMARY KEY,
  instance_id UUID NOT NULL REFERENCES instances(instance_id),
  subject_id TEXT NOT NULL,
  metric TEXT NOT NULL,
  threshold DOUBLE PRECISION NOT NULL,
  observed_value DOUBLE PRECISION NOT NULL,
  severity SMALLINT NOT NULL,             -- 1=INFO, 2=WARN, 3=ERROR, 4=CRITICAL
  action_taken TEXT,
  at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_breaches_instance_at ON boundary_breaches(instance_id, at DESC);
CREATE INDEX idx_breaches_severity_at ON boundary_breaches(severity, at DESC);
```

### 2.7 advisories

```sql
CREATE TABLE advisories (
  id UUID PRIMARY KEY,
  correlation_id UUID NOT NULL,
  schema_version TEXT NOT NULL,
  kind SMALLINT NOT NULL,                 -- enum AdvisoryKind
  subject_id TEXT NOT NULL,
  action TEXT NOT NULL,
  conditions JSONB,
  evidence JSONB NOT NULL,                -- AdvisoryEvidence
  fallback TEXT NOT NULL,
  status TEXT NOT NULL DEFAULT 'pending', -- pending / passed_gate / dismissed / expired / acked
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  expires_at TIMESTAMPTZ NOT NULL,
  acked_at TIMESTAMPTZ,
  ack_user_id TEXT
);

CREATE INDEX idx_advisories_subject_status ON advisories(subject_id, status);
CREATE INDEX idx_advisories_created ON advisories(created_at DESC);
CREATE INDEX idx_advisories_expires ON advisories(expires_at);
```

### 2.8 watchlists / watchlist_items

```sql
CREATE TABLE watchlists (
  id UUID PRIMARY KEY,
  user_id TEXT NOT NULL,
  name TEXT NOT NULL,
  description TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_watchlists_user ON watchlists(user_id);

CREATE TABLE watchlist_items (
  watchlist_id UUID NOT NULL REFERENCES watchlists(id) ON DELETE CASCADE,
  instance_id UUID NOT NULL REFERENCES instances(instance_id),
  subject_id TEXT NOT NULL,
  tags TEXT[],
  added_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  PRIMARY KEY (watchlist_id, instance_id)
);

CREATE INDEX idx_watchlist_items_subject ON watchlist_items(subject_id);
```

### 2.9 notification_logs

```sql
CREATE TABLE notification_logs (
  id UUID PRIMARY KEY,
  user_id TEXT NOT NULL,
  channel TEXT NOT NULL,                  -- push / email / im
  payload JSONB NOT NULL,
  source_advisory_id UUID,
  source_breach_id UUID,
  status TEXT NOT NULL,                   -- sent / failed / digest
  attempted_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  delivered_at TIMESTAMPTZ
) PARTITION BY RANGE (attempted_at);

CREATE INDEX idx_notif_user_attempted ON notification_logs(user_id, attempted_at DESC);
CREATE INDEX idx_notif_status_attempted ON notification_logs(status, attempted_at DESC);
```
