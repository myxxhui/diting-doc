## 三、核心模块设计

### 3.1 事件流消费

```python
# copilot/events/consumer.py

import asyncio
from redis import asyncio as aioredis
from typing import Callable, Dict

class EventConsumer:
    """Redis Stream 事件消费者"""
    
    STREAMS = [
        "events:cryo_guard:reject",
        "events:cryo_guard:degrade",
        "events:cryo_guard:pass",
        "events:thrust:thesis_proposed",
        "events:monitor:health_change",
        "events:exit:sell_signal",
    ]
    
    def __init__(self, redis_url: str):
        self.redis = aioredis.from_url(redis_url)
        self.handlers: Dict[str, Callable] = {}
    
    def register_handler(self, stream: str, handler: Callable):
        """注册事件处理器"""
        self.handlers[stream] = handler
    
    async def consume(self):
        """消费事件流"""
        # 创建消费者组
        for stream in self.STREAMS:
            try:
                await self.redis.xgroup_create(
                    stream, "copilot_group", id="0", mkstream=True
                )
            except Exception:
                pass  # 组已存在
        
        while True:
            # 读取事件
            entries = await self.redis.xreadgroup(
                groupname="copilot_group",
                consumername="copilot_1",
                streams={s: ">" for s in self.STREAMS},
                count=10,
                block=1000,
            )
            
            for stream, messages in entries:
                handler = self.handlers.get(stream)
                if handler:
                    for msg_id, data in messages:
                        await handler(data)
                        await self.redis.xack(stream, "copilot_group", msg_id)
```

### 3.2 持仓体检服务

```python
# copilot/modules/health_check/service.py

from dataclasses import dataclass
from typing import List
from datetime import date, timedelta

@dataclass
class HealthCard:
    symbol: str
    name: str
    color: str         # red/orange/yellow/green
    push_level: int    # 0-3
    health_score: float
    last_change: str   # 变化原因

class HealthCheckService:
    """持仓体检服务"""
    
    def __init__(self, db, redis):
        self.db = db
        self.redis = redis
    
    async def get_dashboard(self, user_id: str) -> dict:
        """获取首屏 4 色卡片"""
        holdings = await self._get_user_holdings(user_id)
        
        cards = {"red": [], "orange": [], "yellow": [], "green": []}
        
        for holding in holdings:
            health = await self._get_latest_health(holding.symbol)
            card = HealthCard(
                symbol=holding.symbol,
                name=holding.name,
                color=self._get_color(health.push_level),
                push_level=health.push_level,
                health_score=health.score,
                last_change=health.change_reason,
            )
            cards[card.color].append(card)
        
        return {
            "cards": cards,
            "summary": {
                "total": len(holdings),
                "red": len(cards["red"]),
                "orange": len(cards["orange"]),
                "yellow": len(cards["yellow"]),
                "green": len(cards["green"]),
            }
        }
    
    async def get_detail(self, symbol: str) -> dict:
        """获取单持仓详情"""
        # 节点 4 态
        state = await self._get_node_state(symbol)
        
        # 健康度曲线 30 天
        history = await self._get_health_history(
            symbol,
            start=date.today() - timedelta(days=30),
            end=date.today()
        )
        
        return {
            "symbol": symbol,
            "state": state,
            "health_history": history,
        }
    
    def _get_color(self, push_level: int) -> str:
        if push_level >= 3:
            return "red"
        elif push_level == 2:
            return "orange"
        elif push_level == 1:
            return "yellow"
        return "green"
```

### 3.3 推荐池服务

```python
# copilot/modules/recommendation/service.py

from dataclasses import dataclass
from typing import List, Optional
from enum import Enum

class ThesisAction(Enum):
    JOIN = "join"           # 加入
    CONSIDER = "consider"   # 待考虑
    NOT_INTERESTED = "not_interested"  # 不感兴趣

@dataclass
class ThesisCard:
    """5 必填元素"""
    thesis_id: str
    symbol: str
    name: str
    
    # 必填 1：观点摘要
    thesis_summary: str
    
    # 必填 2：证据链
    evidence_chain: List[str]
    
    # 必填 3：风险清单
    risks: List[str]
    
    # 必填 4：估值锚点
    valuation_anchor: dict  # {method, target_price, current_pe, ...}
    
    # 必填 5：操作建议
    action: str  # buy/add/watch
    
    # 元信息
    proposed_at: str
    pass_event_id: Optional[str]  # 关联的维度一 PassEvent

class RecommendationService:
    """推荐池服务"""
    
    MAX_WEEKLY_RECOMMENDATIONS = 5
    
    async def get_pool(self, user_id: str) -> List[ThesisCard]:
        """获取当期推荐池（与数据模型中的周期/标签字段对齐，非 L3 step 排期）"""
        # 过滤当期范围内 + 未处理的推荐
        return await self._get_weekly_unhandled(user_id)
    
    async def handle_action(
        self,
        user_id: str,
        thesis_id: str,
        action: ThesisAction
    ):
        """处理用户操作"""
        # 记录用户决策
        await self._save_user_decision(user_id, thesis_id, action)
        
        # 写入归因数据（用于价值账本）
        await self._log_decision_for_attribution(
            user_id, thesis_id, action
        )
    
    async def export_pdf(self, thesis_id: str) -> bytes:
        """导出 PDF"""
        thesis = await self._get_thesis(thesis_id)
        html = await self._render_pdf_template(thesis)
        return self.pdf_generator.generate(html)
    
    def validate_thesis(self, thesis: ThesisCard) -> List[str]:
        """验证 5 必填元素完整性"""
        issues = []
        
        if not thesis.thesis_summary or len(thesis.thesis_summary) < 20:
            issues.append("观点摘要缺失或过短")
        
        if not thesis.evidence_chain or len(thesis.evidence_chain) < 3:
            issues.append("证据链不足 3 条")
        
        if not thesis.risks:
            issues.append("风险清单为空")
        
        if not thesis.valuation_anchor:
            issues.append("估值锚点缺失")
        
        if not thesis.action:
            issues.append("操作建议缺失")
        
        return issues
```

```python
        self.db = db
    
    async def send_alert(self, alert: Alert, user_id: str):
        """发送告警（多通道并行）"""
```

```yaml
version: '3.8'

services:
  copilot:
    build: .
    ports:
      - "8080:8080"
    volumes:
      - ./data:/app/data
    environment:
      - REDIS_URL=redis://redis:6379
      - DATABASE_URL=sqlite:///data/copilot.db
    depends_on:
      - redis
  
  event-consumer:
    build: .
    command: python -m copilot.events.consumer
    environment:
      - REDIS_URL=redis://redis:6379
    depends_on:
      - redis
  
  scheduler:
    build: .
    command: python -m copilot.scheduler
    environment:
      - DATABASE_URL=sqlite:///data/copilot.db
  
  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
```
