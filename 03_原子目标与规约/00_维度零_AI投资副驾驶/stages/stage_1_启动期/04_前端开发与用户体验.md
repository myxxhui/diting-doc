## 一、用户画像与场景

### 1.1 目标用户

| 属性 | 描述 |
|---|---|

### 3.3 实时推送（SSE）

```python
# copilot/api/sse.py

from fastapi import APIRouter
from sse_starlette.sse import EventSourceResponse
import asyncio

router = APIRouter()

@router.get("/events/stream")
async def event_stream(user_id: str):
    """SSE 实时推送"""
    async def generate():
        while True:
            # 检查新告警
            alerts = await get_new_alerts(user_id)
            for alert in alerts:
                yield {
                    "event": "alert",
                    "data": alert.json()
                }
            
            # 检查健康度变化
            changes = await get_health_changes(user_id)
            for change in changes:
                yield {
                    "event": "health_change",
                    "data": change.json()
                }
            
            await asyncio.sleep(5)
    
    return EventSourceResponse(generate())
```

```python
import httpx

class TelegramChannel:
    """Telegram Bot"""
    
    def __init__(self, bot_token: str, chat_id: str):
        self.bot_token = bot_token
```

### 4.3 邮件

```python
# copilot/modules/alert/channels/email.py

import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

class EmailChannel:
    """邮件通道"""
    
    def __init__(self, smtp_host: str, smtp_port: int, username: str, password: str):
        self.smtp_host = smtp_host
        self.smtp_port = smtp_port
        self.username = username
        self.password = password
    
    async def send(self, alert, user_id: str) -> bool:
        """发送邮件"""
        user_email = await get_user_email(user_id)
        
        msg = MIMEMultipart()
        msg['Subject'] = f"[{alert.level.upper()}] {alert.symbol} {alert.alert_type}"
        msg['From'] = self.username
        msg['To'] = user_email
        
        body = f"""
        <h2>{alert.level.upper()} 告警</h2>
        <p><strong>{alert.symbol}</strong> {alert.name}</p>
        <p>类型: {alert.alert_type}</p>
        <p>{alert.message}</p>
        <p>时间: {alert.created_at}</p>
        """
        
        msg.attach(MIMEText(body, 'html'))
        
        try:
            with smtplib.SMTP(self.smtp_host, self.smtp_port) as server:
                server.starttls()
                server.login(self.username, self.password)
                server.send_message(msg)
            return True
        except Exception:
            return False
```

### 5.1 Thesis 卡 PDF

```python
# copilot/modules/recommendation/pdf_generator.py

from weasyprint import HTML, CSS
from jinja2 import Environment, FileSystemLoader

class ThesisPDFGenerator:
    """Thesis 卡 PDF 生成"""
    
    def __init__(self, template_dir: str):
        self.env = Environment(loader=FileSystemLoader(template_dir))
        self.template = self.env.get_template("thesis_pdf.html")
        self.css = CSS(filename=f"{template_dir}/pdf.css")
    
    def generate(self, thesis: dict) -> bytes:
        """生成 PDF"""
        html_content = self.template.render(thesis=thesis)
        html = HTML(string=html_content)
        return html.write_pdf(stylesheets=[self.css])
```
