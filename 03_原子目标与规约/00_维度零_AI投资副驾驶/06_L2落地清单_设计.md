### 3.1 持仓体检报告（module 01）

```typescript
// 数据获取

GET  /api/holdings/{symbol}/health-history?days=30   // 30 天健康度曲线

GET  /api/streams/state-watch?topics=health_change,rebalance_advice
GET  /api/streams/state-watch-exit?topics=sell_signal

// 4 色路由（前端逻辑）
function classifyColor(events): "red" | "orange" | "yellow" | "green" {
  if (events.some(e => e.push_level === "emergency_red" || e.push_level === "regular_red"))
    return "red";
  if (events.some(e => e.push_level === "orange")) return "orange";
  if (events.some(e => e.push_level === "yellow")) return "yellow";
  return "green";
}
```

```typescript
GET  /api/thesis-pool/weekly-pdf?week=2026-W24

// 5 必填强制检查（前端渲染前）
function validateThesisCard(card): boolean {
  return [
    card.logic_chain?.nodes?.length >= 3,
    card.logic_chain?.strong_constraint_count >= 1,
    card.sli_probes?.length > 0,
    card.battlefield && card.window_days,
    card.min_return_threshold && card.target_price,
    card.cognitive_boundary_check?.passed,
  ].every(Boolean);
}
```

```typescript
// 同标的 24h 同类去重
function shouldDedupAlert(newAlert, recentAlerts): boolean {
  return recentAlerts.some(a =>
    a.symbol === newAlert.symbol &&
    a.type === newAlert.type &&
    (now - a.time).hours < 24
  );
```
