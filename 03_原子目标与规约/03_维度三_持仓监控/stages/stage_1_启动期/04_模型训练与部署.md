## 二、叙事一致性模型设计

### 2.1 任务定义

**输入**：
- `thesis`：建仓逻辑摘要（如"公司处于渗透率提升红利期，交付量高增长"）
- `current_facts`：当前事实快照（如"Q1 交付量同比+45%，毛利率 19.2%"）

**输出**：
- `label`：`entailment` / `neutral` / `contradiction`
- `confidence`：置信度（0-1）
- `reasoning`：简短推理说明

```json
{
  "label": "entailment",
  "confidence": 0.92,
  "reasoning": "当前交付量增速 45% 符合高增长预期，毛利率 19.2% 高于 18% 阈值"
}
```

### 2.2 Prompt 模板

```python
NARRATIVE_NLI_PROMPT = """你是一个投资逻辑一致性判断助手。

## 任务
判断「当前事实」是否支持「建仓逻辑」。

## 输入
【建仓逻辑】
{thesis}

【当前事实】
{current_facts}

## 判断标准
- entailment：当前事实明确支持建仓逻辑
- neutral：当前事实与逻辑无明显矛盾，但也不强化
- contradiction：当前事实与建仓逻辑存在明显矛盾

## 输出要求
请以 JSON 格式输出，包含 label、confidence、reasoning 三个字段。

## 输出
"""
```

### 2.3 标签分布目标

| 标签 | 训练集占比 | 说明 |
|---|---|---|
| `entailment` | 40% | 正向支持 |
| `neutral` | 30% | 中性 |
| `contradiction` | 30% | 矛盾 |

## 六、模型评测

### 6.1 评测指标

| 指标 | 目标 | 说明 |
|---|---|---|
| 准确率（Accuracy） | ≥ 0.85 | 3 分类准确率 |
| F1-macro | ≥ 0.80 | 平衡各类别 |
| Entailment Recall | ≥ 0.90 | 不漏正向支持 |
| Contradiction Recall | ≥ 0.85 | 不漏矛盾 |

### 6.2 评测脚本

```python
# training/scripts/evaluate_nli.py

import json
from typing import List, Dict
from sklearn.metrics import classification_report, accuracy_score, f1_score
from transformers import AutoModelForCausalLM, AutoTokenizer
from peft import PeftModel

def load_model(base_path: str, lora_path: str):
    """加载模型"""
    tokenizer = AutoTokenizer.from_pretrained(base_path)
    model = AutoModelForCausalLM.from_pretrained(
        base_path,
        torch_dtype="auto",
        device_map="auto",
    )
    model = PeftModel.from_pretrained(model, lora_path)
    return model, tokenizer

def evaluate(
    model, 
    tokenizer, 
    test_data: List[Dict],
) -> Dict:
    """评测模型"""
    y_true = []
    y_pred = []
    
    for item in test_data:
        # 构造输入
        prompt = NARRATIVE_NLI_PROMPT.format(
            thesis=item["thesis"],
            current_facts=item["current_facts"],
        )
        
        # 推理
        inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
        outputs = model.generate(**inputs, max_new_tokens=200)
        response = tokenizer.decode(outputs[0], skip_special_tokens=True)
        
        # 解析输出
        pred_label = parse_label(response)
        
        y_true.append(item["label"])
        y_pred.append(pred_label)
    
    # 计算指标
    return {
        "accuracy": accuracy_score(y_true, y_pred),
        "f1_macro": f1_score(y_true, y_pred, average="macro"),
        "report": classification_report(y_true, y_pred),
    }

def parse_label(response: str) -> str:
    """解析模型输出中的标签"""
    try:
        # 尝试解析 JSON
        start = response.find("{")
        end = response.rfind("}") + 1
        if start >= 0 and end > start:
            data = json.loads(response[start:end])
            return data.get("label", "neutral")
    except:
        pass
    
    # 降级：关键词匹配
    if "entailment" in response.lower():
        return "entailment"
    elif "contradiction" in response.lower():
        return "contradiction"
    return "neutral"

if __name__ == "__main__":
    # 加载模型
    model, tokenizer = load_model(
        "Qwen/Qwen2.5-7B-Instruct",
        "outputs/narrative_nli_lora_v1",
    )
    
    # 加载测试数据
    with open("training/data/narrative_nli/test.jsonl") as f:
        test_data = [json.loads(line) for line in f]
    
    # 评测
    results = evaluate(model, tokenizer, test_data)
    
    print(f"Accuracy: {results['accuracy']:.4f}")
    print(f"F1-macro: {results['f1_macro']:.4f}")
    print(results['report'])
```

### 7.2 LoRA 热加载

```python
# state_watch/health/narrative_nli.py

from dataclasses import dataclass
from typing import Optional
import httpx

@dataclass
class NLIResult:
    """NLI 结果"""
    label: str                # entailment/neutral/contradiction
    confidence: float         # 0-1
    reasoning: str            # 推理说明

class NarrativeNLI:
    """叙事一致性判断（调用 vLLM）"""
    
    def __init__(self, vllm_url: str = "http://vllm:8000"):
        self.vllm_url = vllm_url
        self.lora_name = "narrative_nli"
    
    async def evaluate(
        self, 
        thesis: str, 
        current_facts: str,
    ) -> NLIResult:
        """
        评估叙事一致性
        """
        prompt = NARRATIVE_NLI_PROMPT.format(
            thesis=thesis,
            current_facts=current_facts,
        )
        
        async with httpx.AsyncClient() as client:
            response = await client.post(
                f"{self.vllm_url}/v1/completions",
                json={
                    "model": self.lora_name,  # 指定 LoRA
                    "prompt": prompt,
                    "max_tokens": 200,
                    "temperature": 0.1,
                },
                timeout=30.0,
            )
            response.raise_for_status()
            data = response.json()
        
        # 解析输出
        text = data["choices"][0]["text"]
        return self._parse_response(text)
    
    def _parse_response(self, text: str) -> NLIResult:
        """解析模型输出"""
        import json
        
        try:
            start = text.find("{")
            end = text.rfind("}") + 1
            if start >= 0 and end > start:
                data = json.loads(text[start:end])
                return NLIResult(
                    label=data.get("label", "neutral"),
                    confidence=data.get("confidence", 0.5),
                    reasoning=data.get("reasoning", ""),
                )
        except:
            pass
        
        # 降级
        return NLIResult(
            label="neutral",
            confidence=0.5,
            reasoning="解析失败，降级为 neutral",
        )
```
