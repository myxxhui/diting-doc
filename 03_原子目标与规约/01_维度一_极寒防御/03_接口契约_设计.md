## 一、契约范围

本文件定义极寒防御模块对外暴露的 5 类核心契约：

| 契约 | 用途 | 协议归属 |
|------|------|---------|
| `RiskEvent` | 风险事件流 | Decision Protocol |
| `GateRequest` / `GateDecision` | 决策门禁判定 | Decision Protocol |
| `BreakerStateChange` | 熔断状态变更通知 | Runtime Protocol |
| `SanitizeRequest` / `SanitizeResult` | 输入污染检测 | Decision Protocol |
| `AuditEntry` | 审计日志写入 | Ops Protocol |

## 二、Proto 定义（v1）

### 2.1 RiskEvent

```proto
syntax = "proto3";
package cryo_guard.v1;

import "google/protobuf/timestamp.proto";
import "google/protobuf/struct.proto";

enum Severity {
  SEVERITY_UNSPECIFIED = 0;
  SEVERITY_INFO = 1;
  SEVERITY_WARN = 2;
  SEVERITY_ERROR = 3;
  SEVERITY_CRITICAL = 4;
}

enum RiskCategory {
  RISK_CATEGORY_UNSPECIFIED = 0;
  RISK_CATEGORY_DATA_QUALITY = 1;       // 数据质量 / 漂移 / 缺失
  RISK_CATEGORY_INFERENCE_SERVICE = 2;  // 推理服务降级 / 失败
  RISK_CATEGORY_EXTERNAL_DEPENDENCY = 3;
  RISK_CATEGORY_INPUT_CONTAMINATION = 4; // prompt injection / 投毒
  RISK_CATEGORY_DECISION_VIOLATION = 5;  // 决策门禁拒绝
  RISK_CATEGORY_AUDIT_INTEGRITY = 6;     // 审计链异常
  RISK_CATEGORY_BUSINESS_BOUNDARY = 7;   // 业务边界（如不允许的标的）
}

message RiskEvent {
  string id = 1;                              // UUID v7
  string correlation_id = 2;                  // 串起一次业务流
  string schema_version = 3;                  // "1.0.0"
  Severity severity = 4;
  RiskCategory category = 5;
  string code = 6;                            // 细分编码，如 "DRIFT_001"
  string message = 7;                         // 人可读说明
  repeated string affected_subjects = 8;      // 标的 / 服务名 / module 名
  string evidence_ref = 9;                    // 证据存储位置（s3://... 或 db://...）
  google.protobuf.Struct extra = 10;          // 自由扩展字段
  string producer = 11;                       // 来源服务
  google.protobuf.Timestamp created_at = 12;
}
```

### 2.2 GateRequest / GateDecision

```proto
syntax = "proto3";
package cryo_guard.v1;

enum GateObjectType {
  GATE_OBJECT_TYPE_UNSPECIFIED = 0;
  GATE_OBJECT_TYPE_RESEARCH_CARD = 1;
  GATE_OBJECT_TYPE_CANDIDATE = 2;
  GATE_OBJECT_TYPE_ADVISORY = 3;          // 调仓 / 退出建议
  GATE_OBJECT_TYPE_EXTERNAL_ACTION = 4;
  GATE_OBJECT_TYPE_KNOWLEDGE_WRITE = 5;
  GATE_OBJECT_TYPE_MODEL_DEPLOY = 6;
}

enum GateResult {
  GATE_RESULT_UNSPECIFIED = 0;
  GATE_RESULT_PASS = 1;
  GATE_RESULT_DEGRADE = 2;                // 允许通过但需降级
  GATE_RESULT_REJECT = 3;
  GATE_RESULT_PENDING_HUMAN = 4;          // 需人审，决议挂起
}

message GateRequest {
  string request_id = 1;
  string correlation_id = 2;
  GateObjectType object_type = 3;
  bytes object_payload = 4;               // 序列化后的对象
  string object_schema_version = 5;
  google.protobuf.Struct context = 6;     // 调用方上下文
  string requester = 7;                   // 调用方服务名
}

message GateLayerResult {
  string layer_name = 1;                  // schema / evidence / compliance / metric / sanitize / breaker / human
  GateResult result = 2;
  string reason = 3;
  google.protobuf.Struct details = 4;
  int32 elapsed_ms = 5;
}

message GateDecision {
  string decision_id = 1;                 // 同 request_id
  GateResult result = 2;                  // 聚合后最严结果
  string reason = 3;
  repeated GateLayerResult layer_results = 4;
  string evidence_ref = 5;                // 证据快照位置
  bytes degraded_payload = 6;             // result=DEGRADE 时返回降级后对象
  google.protobuf.Timestamp decided_at = 7;
}
```

### 2.3 BreakerStateChange

```proto
syntax = "proto3";
package cryo_guard.v1;

enum BreakerState {
  BREAKER_STATE_UNSPECIFIED = 0;
  BREAKER_STATE_CLOSED = 1;     // 正常
  BREAKER_STATE_OPEN = 2;       // 熔断中
  BREAKER_STATE_HALF_OPEN = 3;  // 探测中
}

message BreakerStateChange {
  string service = 1;                     // 服务名
  string caller = 2;                      // 调用方（可选，按调用方维度熔断时填）
  BreakerState from = 3;
  BreakerState to = 4;
  string reason = 5;                      // 触发原因（错误率/延迟/手动）
  google.protobuf.Struct metrics = 6;     // 触发时的关键指标
  google.protobuf.Timestamp changed_at = 7;
}
```

### 2.4 SanitizeRequest / SanitizeResult

```proto
syntax = "proto3";
package cryo_guard.v1;

enum ContaminationKind {
  CONTAMINATION_KIND_UNSPECIFIED = 0;
  CONTAMINATION_KIND_PROMPT_INJECTION = 1;
  CONTAMINATION_KIND_DATA_DRIFT = 2;
  CONTAMINATION_KIND_NUMERIC_ANOMALY = 3;
  CONTAMINATION_KIND_OVERSIZED = 4;
  CONTAMINATION_KIND_ENCODING = 5;
  CONTAMINATION_KIND_TIMESTAMP = 6;
  CONTAMINATION_KIND_UNTRUSTED_URL = 7;
}

message Finding {
  ContaminationKind kind = 1;
  Severity severity = 2;
  string detail = 3;
  google.protobuf.Struct evidence = 4;
}

message SanitizeRequest {
  string content_type = 1;                // text / numeric / json / mixed
  bytes payload = 2;
  string source = 3;
  google.protobuf.Struct context = 4;
}

message SanitizeResult {
  bool passed = 1;
  repeated Finding findings = 2;
  bytes sanitized_payload = 3;            // 可选，passed=false 时尝试给出脱敏版本
}
```

### 2.5 AuditEntry

```proto
syntax = "proto3";
package cryo_guard.v1;

message AuditEntry {
  string entry_id = 1;
  string decision_id = 2;                 // 关联 GateDecision
  string actor = 3;                       // 服务名 / 用户 ID
  string action = 4;                      // gate_decision / human_review / breaker_change / external_action
  bytes payload = 5;                      // 原始请求 + 决议
  string payload_hash = 6;                // sha256
  string previous_hash = 7;               // 链式 hash
  string signature = 8;                   // 节点签名（可选；快照时统一签）
  google.protobuf.Timestamp ts = 9;
}
```

## 三、HTTP / gRPC 端点

| 端点 | 方法 | 说明 |
|------|------|------|
| `POST /v1/risk-events` | gRPC + REST | 提交风险事件 |
| `GET  /v1/risk-events` | REST | 检索（支持 severity / category / time_range / actor） |
| `POST /v1/gate/decisions` | gRPC + REST | 同步门禁判定 |
| `GET  /v1/gate/decisions/{id}` | REST | 查询决策详情 + 回放上下文 |
| `POST /v1/sanitize` | gRPC + REST | 输入污染检测 |
| `GET  /v1/breakers` | REST | 列出熔断器状态 |
| `GET  /v1/breakers/{service}` | REST | 单服务熔断状态 |
| `POST /v1/breakers/{service}/reset` | REST | 手动重置（需高权限 + 审计） |
| `POST /v1/audit/entries` | gRPC | 写审计 |
| `GET  /v1/audit/entries` | REST | 检索审计 |
| `POST /v1/audit/exports` | REST | 触发合规导出 |

## 四、版本兼容策略

- 协议版本号：`schema_version` 字段（语义化版本 `MAJOR.MINOR.PATCH`）
- **MAJOR 变更**：必须新建 `cryo_guard.v2`，旧版至少维护 6 个月
- **MINOR 变更**：仅新增可选字段，旧客户端不受影响
- **PATCH 变更**：bug 修复 / 文档更新

## 五、对外通用错误码

| 错误码 | HTTP | 含义 |
|--------|------|------|
| `CG.INVALID_PAYLOAD` | 400 | payload 无法反序列化或缺必填字段 |
| `CG.RULE_LOAD_FAILED` | 503 | 门禁规则加载失败，全部请求拒绝 |
| `CG.AUDIT_WRITE_FAILED` | 500 | 审计写失败（业务必须重试） |
| `CG.SERVICE_BREAKER_OPEN` | 503 | 目标服务被熔断，请稍后重试 |
| `CG.HUMAN_REVIEW_REQUIRED` | 202 | 决议挂起，需人审；返回 polling URL |
| `CG.UNAUTHORIZED` | 401 | 调用方无权限 |
