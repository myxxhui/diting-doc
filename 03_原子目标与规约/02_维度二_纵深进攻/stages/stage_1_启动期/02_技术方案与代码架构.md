## 三、核心模块

### 3.1 剧本基类

```python
# deep_strike/playbooks/base_playbook.py

from abc import ABC, abstractmethod
from pydantic import BaseModel

class PlaybookResult(BaseModel):
    decision: str  # propose/watch/discard
    confidence: float
    thesis_card: dict | None
    evidence: list[str]

class BasePlaybook(ABC):
    @abstractmethod
    async def scan(self, symbol: str) -> PlaybookResult:
        """扫描单个标的"""
        pass
    
    @abstractmethod
    async def batch_scan(self, symbols: list[str]) -> list[PlaybookResult]:
        """批量扫描"""
        pass
```

### 3.2 利润截留扫描仪

```python
# deep_strike/playbooks/profit_capture.py

class ProfitCapturePlaybook(BasePlaybook):
    SIGNALS = {
        "gross_margin_qoq_up": 0.3,
        "cost_growth_below_revenue": 0.25,
        "operating_leverage": 0.25,
        "receivable_turnover_up": 0.1,
        "inventory_turnover_up": 0.1,
    }
    
    async def scan(self, symbol: str) -> PlaybookResult:
        # 1. 获取财务数据
        data = await self.fetch_financial_data(symbol)
        
        # 2. 计算信号
        signals = self.calculate_signals(data)
        
        # 3. 计算置信度
        confidence = self.calculate_confidence(signals)
        
        # 4. 生成 thesis 卡片（如果 propose）
        if confidence >= 0.7:
            thesis = await self.generate_thesis(symbol, data, signals)
            return PlaybookResult(
                decision="propose",
                confidence=confidence,
                thesis_card=thesis,
                evidence=signals["triggered"]
            )
        elif confidence >= 0.4:
            return PlaybookResult(decision="watch", confidence=confidence, ...)
        else:
            return PlaybookResult(decision="discard", confidence=confidence, ...)
```

### 3.3 Thesis 卡片生成器

```python
# deep_strike/engines/thesis_generator.py

class ThesisCardGenerator:
    REQUIRED_ELEMENTS = [
        "thesis_summary",
        "evidence_chain",
        "risks",
        "valuation_anchor",
        "action"
    ]
    
    async def generate(self, symbol: str, data: dict, signals: dict) -> dict:
        # 1. 构建证据链
        evidence = await self.evidence_builder.build(symbol, data)
        
        # 2. LLM 生成观点摘要
        summary = await self.llm.generate_summary(symbol, evidence)
        
        # 3. LLM 识别风险
        risks = await self.llm.identify_risks(symbol, data)
        
        # 4. 计算估值锚点
        valuation = self.calculate_valuation(data)
        
        # 5. 确定操作建议
        action = self.determine_action(signals)
        
        card = {
            "thesis_summary": summary,
            "evidence_chain": evidence,
            "risks": risks,
            "valuation_anchor": valuation,
            "action": action
        }
        
        # 校验完整性
        self.validate(card)
        return card
```

## 四、API 设计

```yaml
# 扫描单个标的
POST /api/playbooks/{playbook_id}/scan
  Request: {symbol: str}
  Response: PlaybookResult

# 批量扫描
POST /api/playbooks/{playbook_id}/batch-scan
  Request: {symbols: list[str]}
  Response: list[PlaybookResult]

# 获取 thesis 卡片
GET /api/thesis/{thesis_id}
  Response: ThesisCard

# 人工确认
POST /api/thesis/{thesis_id}/confirm
  Request: {action: "confirm" | "reject"}
  Response: {success: bool}
```

-- Thesis 卡片
CREATE TABLE thesis_cards (
    id INTEGER PRIMARY KEY,
    thesis_id TEXT UNIQUE,
    symbol TEXT,
    playbook_id TEXT,

    thesis_summary TEXT,
    evidence_chain JSON,
    risks JSON,
    valuation_anchor JSON,
    action TEXT,
    status TEXT,  -- proposed/confirmed/rejected

CREATE TABLE scan_logs (
    id INTEGER PRIMARY KEY,
    playbook_id TEXT,
    symbol TEXT,
    decision TEXT,
    confidence REAL,

);
